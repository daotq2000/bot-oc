**Strategy Calculator Logic**:
```typescript
interface StrategyCalculation {
  openPrice: number;
  oc: number;
  extend: number;
  amount: number;
  takeProfit: number;
  
  // Calculated fields
  longEntry: number;
  longTP: number;
  longProfit: number;
  shortEntry: number;
  shortTP: number;
  shortProfit: number;
}

function calculateStrategy(params: StrategyCalculation) {
  const ocDecimal = params.oc / 100;
  const extendDecimal = params.extend / 100;
  const tpDecimal = (params.oc * params.takeProfit) / 100000;
  
  // LONG calculations
  const longEntry = params.openPrice * (1 - ocDecimal * extendDecimal);
  const longTP = longEntry * (1 + tpDecimal);
  const longProfit = params.amount * tpDecimal;
  
  // SHORT calculations
  const shortEntry = params.openPrice * (1 + ocDecimal * extendDecimal);
  const shortTP = shortEntry * (1 - tpDecimal);
  const shortProfit = params.amount * tpDecimal;
  
  return {
    longEntry,
    longTP,
    longProfit,
    shortEntry,
    shortTP,
    shortProfit
  };
}
```

    queryFn: api.getBots,
    refetchInterval: 30000, // Refetch every 30s
  });

  return useQuery({
    queryKey: ['bots', id],
    queryFn: () => api.getBot(id),
    enabled: !!id,
  });
}

export function useCreateBot() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: api.createBot,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['bots'] });
      toast.success('Bot created successfully!');
    },
    onError: (error) => {
      toast.error(`Failed to create bot: ${error.message}`);
    },
  });
}

      queryClient.invalidateQueries({ queryKey: ['bots'] });

      toast.success('Bot updated successfully!');
    },
  });
}
